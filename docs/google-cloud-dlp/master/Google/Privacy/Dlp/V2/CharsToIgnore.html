class CharsToIgnore
  module CommonCharsToIgnore
    COMMON_CHARS_TO_IGNORE_UNSPECIFIED = 0

    # 0-9
    NUMERIC = 1

    # A-Z
    ALPHA_UPPER_CASE = 2

    # a-z
    ALPHA_LOWER_CASE = 3

    # US Punctuation, one of !"#$%&'()*+,-./:;<=>?@[\]^_+{|}~
    PUNCTUATION = 4

    # Whitespace character, one of [ \t\n\x0B\f\r]
    WHITESPACE = 5
  end
end
