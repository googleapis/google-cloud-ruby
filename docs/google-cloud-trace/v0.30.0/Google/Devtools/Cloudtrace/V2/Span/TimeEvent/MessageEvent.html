class MessageEvent
  # Indicates whether the message was sent or received.
  module Type
    # Unknown event type.
    TYPE_UNSPECIFIED = 0

    # Indicates a sent message.
    SENT = 1

    # Indicates a received message.
    RECEIVED = 2
  end
end
