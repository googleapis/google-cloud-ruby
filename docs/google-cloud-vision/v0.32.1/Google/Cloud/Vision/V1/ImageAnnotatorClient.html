Overview

Service that performs Google Cloud Vision API detection tasks over client images, such as face, landmark, logo, label, and text detection. The ImageAnnotator service returns detected entities from the images.

  end

def crop_hints_detection \
  images: [], 
  image: nil,
  max_results: 10,
  options: nil,
  async: false,
  mime_type: nil,
  batch_size: 10,
  destination: nil,
  &blk

  feature = { type: :CROP_HINTS }
  feature[:max_results] = max_results
  images << image if image

  formatted_images = images.map do |img|
    formatted_image = normalize_image img
    formatted_image[:mime_type] = mime_type if mime_type
    formatted_image
  end

  requests = formatted_images.map do |img|
    {
      image: img,
      features: [feature]
    }
  end

  if async
    requests.map! do |request|
      {
        input_config: {
          gcs_source: {
            uri: request[:image][:source][:gcs_image_uri]
          },
          mime_type: mime_type
        },
        features: request[:features],
        output_config: {
          gcs_destination: {
            uri: destination
          },
          batch_size: batch_size
        }
      }
    end
    async_batch_annotate_files requests, options: options
  else
    batch_annotate_images requests, options: options, &blk
  end
end

def text_detection \
  images: [], 
  image: nil,
  max_results: 10,
  options: nil,
  async: false,
  mime_type: nil,
  batch_size: 10,
  destination: nil,
  &blk

  feature = { type: :TEXT_DETECTION }
  feature[:max_results] = max_results
  images << image if image

  formatted_images = images.map do |img|
    formatted_image = normalize_image img
    formatted_image[:mime_type] = mime_type if mime_type
    formatted_image
  end

  requests = formatted_images.map do |img|
    {
      image: img,
      features: [feature]
    }
  end

  if async
    requests.map! do |request|
      {
        input_config: {
          gcs_source: {
            uri: request[:image][:source][:gcs_image_uri]
          },
          mime_type: mime_type
        },
        features: request[:features],
        output_config: {
          gcs_destination: {
            uri: destination
          },
          batch_size: batch_size
        }
      }
    end
    async_batch_annotate_files requests, options: options
  else
    batch_annotate_images requests, options: options, &blk
  end
end
